```js
/**
 * @param {string} digits
 * @return {string[]}
 */
var letterCombinations = function (digits) {
  if (digits.length === 0) {
    return [];
  }

  const digitToLetters = {
    2: 'abc',
    3: 'def',
    4: 'ghi',
    5: 'jkl',
    6: 'mno',
    7: 'pqrs',
    8: 'tuv',
    9: 'wxyz',
  };

  const combinations = [''];
  for (let i = 0; i < digits.length; i++) {
    const digit = digits[i]; // 2
    const letters = digitToLetters[digit]; // abc

    const tempCombinations = [];
    for (let j = 0; j < combinations.length; j++) {
      const combination = combinations[j];

      for (let k = 0; k < letters.length; k++) {
        tempCombinations.push(combination + letters[k]);
      }
    }
    combinations.length = 0;
    combinations.push(...tempCombinations);
  }
  return combinations;
};
```

```js
/**
 * @param {string} digits
 * @return {string[]}
 */
var letterCombinations = function (digits) {
  if (digits.length === 0) {
    return [];
  }

  const digitToLetters = {
    2: 'abc',
    3: 'def',
    4: 'ghi',
    5: 'jkl',
    6: 'mno',
    7: 'pqrs',
    8: 'tuv',
    9: 'wxyz',
  };

  const combinations = [];

  function backtrack(index, combination) {
    if (index === digits.length) {
      combinations.push(combination);
      return;
    }

    const digit = digits[index];
    const letters = digitToLetters[digit];

    for (let i = 0; i < letters.length; i++) {
      backtrack(index + 1, combination + letters[i]);
    }
  }

  backtrack(0, '');

  return combinations;
};
```

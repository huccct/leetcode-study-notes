```js
/**
 * @param {number[]} nums
 * @return {number[][]}
 */
var subsets = function (nums) {
  let res = [];

  const backtrack = (start, subset) => {
    res.push(subset.slice());

    for (let i = start; i < nums.length; i++) {
      subset.push(nums[i]);
      backtrack(i + 1, subset);
      subset.pop();
    }
  };

  backtrack(0, []);
  return res;
};
```

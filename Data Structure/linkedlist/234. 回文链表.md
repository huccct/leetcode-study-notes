### **思路：**

### **代码:**
**方法一：**
```js
/**
 * Definition for singly-linked list.
 * function ListNode(val, next) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.next = (next===undefined ? null : next)
 * }
 */
/**
 * @param {ListNode} head
 * @return {boolean}
 */
var isPalindrome = function(head) {
    const arr = [];
    while(head) {
        arr.push(head.val);
        head = head.next;
    }

    for(let i = 0, j = arr.length - 1; i < arr.length; i++, j--) {
        if(arr[i] !== arr[j]) {
            return false;
        }
    }
    return true;
};
```

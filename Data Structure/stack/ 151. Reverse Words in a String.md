**Method One:**

```js
/**
 * @param {string} s
 * @return {string}
 */
var reverseWords = function (s) {
  return s.trim().split(/\s+/).reverse().join(' ');
};
```

**Method Two:**

```js
/**
 * @param {string} s
 * @return {string}
 */
var reverseWords = function (s) {
  const stk = [];
  let word = '';

  for (let i = 0; i < s.length; i++) {
    if (s[i] !== ' ') {
      word += s[i];
    } else if (word !== '') {
      stk.push(word);
      word = '';
    }
  }

  if (word !== '') {
    stk.push(word);
  }

  return stk.reverse().join(' ');
};
```
